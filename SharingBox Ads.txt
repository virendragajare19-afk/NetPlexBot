    <script type="text/javascript">
        // --- YAHAN APNE DUSRE (STREAM/DOWNLOAD) BLOGGER PAGE KA URL DAALEIN ---
        var streamPageUrl = "https://srgboxstrm.blogspot.com";
        // -----------------------------------------------------------------

        function getParameterByName(name, url) {
            // (Yeh function bilkul sahi hai)
            if (!url) url = window.location.href; name = name.replace(/[\[\]]/g, '\\$&');
            var regex = new RegExp('[?&]' + name + '(=([^&#]*)|&|#|$)'), results = regex.exec(url);
            if (!results) return null; if (!results[2]) return '';
            return decodeURIComponent(results[2].replace(/\+/g, ' '));
        }

        window.onload = function() {
            const uniqueId = getParameterByName('id');
            if (!uniqueId) {
                document.body.innerHTML = "<h1 style='text-align: center; margin-top: 50px;'>ERROR: Invalid or Missing Link ID.</h1>";
                return;
            }
            const timerButton = document.getElementById('timer-button');
            const continueButton = document.getElementById('continue-button');
            const countdownSpan = document.getElementById('countdown');
            const getLinkSection = document.getElementById('get-link-section');
            const getLinkButton = document.getElementById('get-link-button');
            let countdown = 10;
            const timerInterval = setInterval(function() {
                countdown--;
                countdownSpan.textContent = countdown;
                if (countdown <= 0) {
                    clearInterval(timerInterval);
                    timerButton.style.display = 'none';
                    continueButton.style.display = 'inline-block';
                }
            }, 1000);
            
            continueButton.addEventListener('click', function(event) {
                event.preventDefault();
                // --- YAHAN BADLAV KIYA GAYA HAI: Ab yeh 'display' ko badlega ---
                getLinkSection.style.display = 'block'; // 'visibility' ke bajaye 'display'
                document.getElementById('get-link-section').scrollIntoView({ behavior: 'smooth' });
            });
            
            getLinkButton.addEventListener('click', function(event) {
                event.preventDefault();
                const finalUrl = streamPageUrl + '?id=' + uniqueId;
                window.location.replace(finalUrl);
            });
        };
    </script>
