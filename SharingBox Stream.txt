    <script>
        // === YEH NAYA, SAHI WALA JAVASCRIPT HAI ===
        const RENDER_APP_URL = "https://sharingbox.onrender.com";
        const BOT_USERNAME = "SharingBoxBot";
        document.getElementById('upload-btn').href = `https://t.me/${BOT_USERNAME}`;

        // This function runs immediately, as soon as the script is loaded
        (function initPage() {
            const homepageScreen = document.getElementById('homepage-screen');
            const downloadScreen = document.getElementById('download-screen');
            const body = document.body;
            const header = document.getElementById('header');
            const footer = document.getElementById('footer');
            const params = new URLSearchParams(window.location.search);
            const fileId = params.get('id');

            if (fileId) {
                // --- AGAR URL MEIN ID HAI ---
                // Immediately hide homepage, show download page, and change theme
                homepageScreen.style.display = 'none';
                downloadScreen.style.display = 'block';
                body.className = 'flex flex-col min-h-screen theme-dark';
                header.className = 'w-full bg-gray-800 shadow-lg p-4 text-white';
                footer.className = 'w-full bg-gray-800 mt-auto p-4 text-center text-gray-400 text-sm';
                
                // Now start fetching the data
                fetchFileData(fileId);
            } else {
                // --- AGAR URL MEIN ID NAHI HAI ---
                // Do nothing. The page is already showing the homepage correctly.
                document.title = "Sharing Box - Your Secure File Sharing Hub";
            }
        })();
        
        async function fetchFileData(fileId) {
            const loaderContainer = document.getElementById('loader-container');
            const contentContainer = document.getElementById('content-container');
            const fileNameEl = document.getElementById('file-name');
            const fileSizeEl = document.getElementById('file-size');
            const buttonContainer = document.getElementById('button-container');
            
            try {
                const response = await fetch(`${RENDER_APP_URL}/api/file/${fileId}`);
                if (!response.ok) { throw new Error(`Server error (${response.status})`); }
                const data = await response.json();
                fileNameEl.innerText = data.file_name;
                fileSizeEl.innerText = `File Size: ${data.file_size}`;
                document.title = `Download ${data.file_name}`;
                const svg = '<svg xmlns="http://www.w3.org/2000/svg" class="h-5 w-5 mr-2" viewBox="0 0 20 20" fill="currentColor"><path fill-rule="evenodd" d="M3 17a1 1 0 011-1h12a1 1 0 110 2H4a1 1 0 01-1-1zm3.293-7.707a1 1 0 011.414 0L9 10.586V3a1 1 0 112 0v7.586l1.293-1.293a1 1 0 111.414 1.414l-3 3a1 1 0 01-1.414 0l-3-3a1 1 0 010-1.414z" clip-rule="evenodd" /></svg>';
                const mimeType = data.mime_type || 'video/*';
                const mxPlayerLink = `intent:${data.direct_dl_link}#Intent;action=android.intent.action.VIEW;type=${mimeType};end`;
                const vlcIntentLink = `intent:${data.direct_dl_link}#Intent;action=android.intent.action.VIEW;type=${mimeType};package=org.videolan.vlc;end`;

                buttonContainer.innerHTML = '';
                if (data.is_media) {
                    buttonContainer.innerHTML = `<a href="${data.direct_dl_link}" class="btn btn-primary">${svg} Direct Download</a><a href="${mxPlayerLink}" class="btn btn-secondary">Play in MX Player</a><a href="${vlcIntentLink}" class="btn btn-secondary">Play in VLC</a>`;
                } else {
                    buttonContainer.innerHTML = `<a href="${data.direct_dl_link}" class="btn btn-primary w-full sm:w-auto">${svg} Download</a>`;
                }

                loaderContainer.style.display = 'none';
                contentContainer.style.display = 'block';
            } catch (error) {
                fileNameEl.innerText = "Error fetching file details.";
                fileSizeEl.innerText = "Please try again later.";
                loaderContainer.style.display = 'none';
                contentContainer.style.display = 'block';
                console.error(error);
            }
        }
    </script>
